 function load(file) {
	var xhr = new XMLHttpRequest();
	xhr.open('GET', '../core/popup/' + file, true);
	xhr.onreadystatechange = function() {
		if(xhr.readyState !== 4) return;
		document.write(xhr.responseText);
		document.close();
	};
	xhr.send();
}
